# EMPTY
EMPTY(NIL) = TRUE;
EMPTY(CONS(x_ y_)) = FALSE;

# EQUAL
EQUALB(x_ x_) = TRUE;
EQUALB(x_ y_) = FALSE;

EQUALN(x_ x_) = TRUE;
EQUALN(x_ y_) = FALSE;

EQUALL(x_ x_) = TRUE;
EQUALL(x_ y_) = FALSE;

# HEAD
HEAD(NIL) = NAN;
HEAD(CONS(x_ y_)) = x_;

# IF
IFB(TRUE  x_ y_) = x_;
IFB(FALSE x_ y_) = y_;

IFN(TRUE  x_ y_) = x_;
IFN(FALSE x_ y_) = y_;

IFL(TRUE  x_ y_) = x_;
IFL(FALSE x_ y_) = y_;

# TAIL
TAIL(NIL) = NIL;
TAIL(CONS(x_ y_)) = y_;
